let mouseX, mouseY

let halfWidth  = canvas.width/2
let halfHeight = canvas.height/2

const amp = 20

const getMousePos = e => {
  const rect = canvas.getBoundingClientRect()

  mouseX = e.clientX - rect.left
  mouseY = e.clientY - rect.top
}

document.body.addEventListener('mousemove', getMousePos, false)

// requestAnimationFrame ->
const x = halfWidth + ((mouseX - halfWidth) / halfWidth) * amp
const y = halfHeight + ((mouseY - halfHeight) / halfHeight) * amp
